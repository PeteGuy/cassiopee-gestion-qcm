\element{devel-seance1}{
\begin{questionmult}{qu-seance1-a-0}
  Parmi les égalités ci-dessous, indiquer celles qui sont vraies
  ($\star$ est le produit de convolution, $\delta(.)$ la
  distribution de Dirac, $t$ une variable muette, $a$ une
  constante).
\begin{multicols}{2}
  \begin{reponses}
    \bonne{$\delta(t-a)\star x(t) = \delta(t)\star x(t-a)$ }
    \bonne{$\delta(t-a)\star x(t) = x(t)\star \delta(t-a)$ }
    \bonne{$x(t)\star\delta(t-a) = x(t-a)$}
    \bonne{$x(t)\star\delta(t+a) = x(t+a)$}
    \mauvaise{ $\delta(t-a)\star x(t) = \delta(t)\star x(t+a)$ }
    \mauvaise{$\delta(t-a)\star x(t) = x(t)\star \delta(t+a)$ }
    \mauvaise{$x(t)\star\delta(t-a) = x(t+a)$}
    \mauvaise{$x(t)\star\delta(t-a) = x(at)$}
    \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(at)$}
    \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(\frac{t}{a})$}
    \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(at)$ pour
$a\neq 0$.}
    \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(\frac{t}{a})$
pour $a\neq 0$.}
    \mauvaise{$x(t)\star\delta(t+a) = \delta(t+a)$}
    \mauvaise{$x(t)\star\delta(t+a) = x(t) + \delta(a)$}
    \mauvaise{$x(t+a)\star\delta(t) = x(t) + \delta(a)$}
  \end{reponses}
\end{mutlicols}
\end{questionmult}
}
