\element{devel-seance1}{
  \begin{questionmult}{qu-seance1-a-0}%selection complete
    Parmi les égalités ci-dessous, indiquer celles qui sont vraies
    ($\star$ est le produit de convolution, $\delta(.)$ la
    distribution de Dirac, $t$ une variable muette, $a$ une
    constante).
    \begin{multicols}{2}
      \begin{reponses}
      \bonne{$\delta(t-a)\star x(t) = \delta(t)\star x(t-a)$ }
      \bonne{$\delta(t-a)\star x(t) = x(t)\star \delta(t-a)$ }
      \bonne{$x(t)\star\delta(t-a) = x(t-a)$}
      \bonne{$x(t)\star\delta(t+a) = x(t+a)$}

      \mauvaise{ $\delta(t-a)\star x(t) = \delta(t)\star x(t+a)$ }
      \mauvaise{$\delta(t-a)\star x(t) = x(t)\star \delta(t+a)$ }
      
      \mauvaise{$x(t)\star\delta(t-a) = x(t+a)$}
      \mauvaise{$x(t)\star\delta(t-a) = x(at)$}
      
      \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(at)$}
      \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(\frac{t}{a})$}
      \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(at)$ pour
        $a\neq 0$.}
      \mauvaise{$x(t)\star\delta(t-a) = \frac{1}{|a|}x(\frac{t}{a})$
        pour $a\neq 0$.}
      
      \mauvaise{$x(t)\star\delta(t+a) = \delta(t+a)$}
      \mauvaise{$x(t)\star\delta(t+a) = x(t) + \delta(a)$}
      \mauvaise{$x(t+a)\star\delta(t) = x(t) + \delta(a)$}
    \end{reponses}
  \end{multicols}
\end{questionmult}
}

\element{devel-seance1}{
  \begin{questionmult}{qu-seance1-b-0}%selection complete
    Parmi les égalités ci-dessous, indiquer celles qui sont vraies
    ($t$ est la variable muette de dépendance des signaux et $f$ la
    variable muette des fréquences correspondantes, $x(t) \TF X(f)$
    indique que $X(f)$ est la transformée de Fourier de $x(t)$, $a$ et
    $L$ sont des constantes).
    \begin{reponses}
      \bonne{$\delta(t-a) \TF e^{-\ic 2\pi f a}$ }
      \mauvaise{$\delta(t-a) \TF e^{+\ic 2\pi f a}$ }
      \bonne{$\cos(2\pi a t) \TF \frac{1}{2}\Big[\delta(f-a) +
        \delta(f+a)\Big]$ }
      \mauvaise{$\cos(2\pi a t) \TF \frac{1}{2\ic}\Big[\delta(f-a) -
        \delta(f+a)\Big]$ }
      \mauvaise{$\cos(2\pi a t) \TF \delta(f-a) + \ic \delta(f+a)$ }
      \bonne{$\cos(2\pi a t)\indic_{[-L,L]}(t) \TF L\Big[ \sinc(2\pi
        L(f-a)) + \sinc(2\pi L(f+a)) \Big]$}
      \mauvaise{$\cos(2\pi a t)\indic_{[-L,L]}(t) \TF L \sinc(2\pi
        L(f-a))$}
      \bonne{$\sin(2\pi a t) \TF \frac{1}{2\ic}\Big[\delta(f-a) -
        \delta(f+a)\Big]$ }
      \mauvaise{$\sin(2\pi a t) \TF \frac{1}{2}\Big[\delta(f-a) +
        \delta(f+a)\Big]$ }
      \mauvaise{$\sin(2\pi a t) \TF \delta(f-a) + \ic\delta(f+a)$ }
      \mauvaise{$\sin(2\pi a t)\indic_{[-L,L]}(t) \TF \ic L\Big[
        \sinc(2\pi L(f+a)) - \sinc(2\pi L(f+a)) \Big]$} %% post modif mauvaise
      \mauvaise{$\sin(2\pi a t)\indic_{[-L,L]}(t) \TF L \sinc(2\pi
        L(f-a))$}
    \end{reponses}
\end{questionmult}
}
